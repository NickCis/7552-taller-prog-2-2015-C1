371  char *p = NULL;
372

374  assert(io->len <= io->size);
375 
376  /* check overflow */
377  if (len > ~(size_t)0 - (size_t)(io->buf + io->len)) {
378  return 0;
379  }
380 
381  if (len <= 0) {
382  } else if (io->len + len <= io->size) {
383  memcpy(io->buf + io->len, buf, len);
384  io->len += len;
385  } else if ((p = (char *) NS_REALLOC(io->buf, io->len + len)) != NULL) {
386  io->buf = p;
387  memcpy(io->buf + io->len, buf, len);
388  io->len += len;
389  io->size = io->len;
390  } else {
391  len = 0;
392  }

451 // Print message to buffer. If buffer is large enough to hold the message,
452 // return buffer. If buffer is to small, allocate large enough buffer on heap,
453 // and return allocated buffer.
454 int ns_avprintf(char **buf, size_t size, const char *fmt, va_list ap) {
455  va_list ap_copy;
456  int len;
457 
458  va_copy(ap_copy, ap);
459  len = vsnprintf(*buf, size, fmt, ap_copy);
460  va_end(ap_copy);
461 
462  if (len < 0) {
463  // eCos and Windows are not standard-compliant and return -1 when
464  // the buffer is too small. Keep allocating larger buffers until we
465  // succeed or out of memory.
466  *buf = NULL;
467  while (len < 0) {
468  if (*buf) NS_FREE(*buf);
469  size *= 2;
470  if ((*buf = (char *) NS_MALLOC(size)) == NULL) break;
471  va_copy(ap_copy, ap);
472  len = vsnprintf(*buf, size, fmt, ap_copy);
473  va_end(ap_copy);
474  }
475  } else if (len > (int) size) {
476  // Standard-compliant code path. Allocate a buffer that is large enough.
477  if ((*buf = (char *) NS_MALLOC(len + 1)) == NULL) {
478  len = -1;
479  } else {
480  va_copy(ap_copy, ap);
481  len = vsnprintf(*buf, len + 1, fmt, ap_copy);
482  va_end(ap_copy);
483  }
484  }
485 
486  return len;
487 }

512 static void hexdump(struct ns_connection *nc, const char *path,
513  int num_bytes, int ev) {
514  const struct iobuf *io = ev == NS_SEND ? &nc->send_iobuf : &nc->recv_iobuf;
515  FILE *fp;
516  char *buf, src[60], dst[60];
517  int buf_size = num_bytes * 5 + 100;
518 
519  if ((fp = fopen(path, "a")) != NULL) {
520  ns_sock_to_str(nc->sock, src, sizeof(src), 3);
521  ns_sock_to_str(nc->sock, dst, sizeof(dst), 7);
522  fprintf(fp, "%lu %p %s %s %s %d\n", (unsigned long) time(NULL),
523  nc->user_data, src,
524  ev == NS_RECV ? "<-" : ev == NS_SEND ? "->" :
525  ev == NS_ACCEPT ? "<A" : ev == NS_CONNECT ? "C>" : "XX",
526  dst, num_bytes);

528  ns_hexdump(io->buf + (ev == NS_SEND ? 0 : io->len) -
529  (ev == NS_SEND ? 0 : num_bytes), num_bytes, buf, buf_size);
530  fprintf(fp, "%s", buf);
531  NS_FREE(buf);
532  }
533  fclose(fp);
534  }

587 int ns_socketpair2(sock_t sp[2], int sock_type) {
588  union socket_address sa;
589  sock_t sock;
590  socklen_t len = sizeof(sa.sin);
591  int ret = 0;
592 
593  sp[0] = sp[1] = INVALID_SOCKET;
594 
595  (void) memset(&sa, 0, sizeof(sa));
596  sa.sin.sin_family = AF_INET;
597  sa.sin.sin_port = htons(0);
598  sa.sin.sin_addr.s_addr = htonl(0x7f000001);
599 
600  if ((sock = socket(AF_INET, sock_type, 0)) != INVALID_SOCKET &&
601  !bind(sock, &sa.sa, len) &&
602  (sock_type == SOCK_DGRAM || !listen(sock, 1)) &&
603  !getsockname(sock, &sa.sa, &len) &&
604  (sp[0] = socket(AF_INET, sock_type, 0)) != INVALID_SOCKET &&
605  !connect(sp[0], &sa.sa, len) &&
606  (sock_type == SOCK_STREAM ||
607  (!getsockname(sp[0], &sa.sa, &len) && !connect(sock, &sa.sa, len))) &&
608  (sp[1] = (sock_type == SOCK_DGRAM ? sock :
609  accept(sock, &sa.sa, &len))) != INVALID_SOCKET) {
610  ns_set_close_on_exec(sp[0]);
611  ns_set_close_on_exec(sp[1]);
612  ret = 1;

670 // Address format: [PROTO://][IP_ADDRESS:]PORT[:CERT][:CA_CERT]
671 static int ns_parse_address(const char *str, union socket_address *sa,
672  int *proto, int *use_ssl, char *cert, char *ca) {
673  unsigned int a, b, c, d, port;
674  int n = 0, len = 0;
675  char host[200];
676 #ifdef NS_ENABLE_IPV6
677  char buf[100];
678 #endif
679 
680  // MacOS needs that. If we do not zero it, subsequent bind() will fail.
681  // Also, all-zeroes in the socket address means binding to all addresses
682  // for both IPv4 and IPv6 (INADDR_ANY and IN6ADDR_ANY_INIT).
683  memset(sa, 0, sizeof(*sa));
684  sa->sin.sin_family = AF_INET;
685 
686  *proto = SOCK_STREAM;
687  *use_ssl = 0;
688  cert[0] = ca[0] = '\0';
689 
690  if (memcmp(str, "ssl://", 6) == 0) {
691  str += 6;
692  *use_ssl = 1;
693  } else if (memcmp(str, "udp://", 6) == 0) {
694  str += 6;
695  *proto = SOCK_DGRAM;
696  } else if (memcmp(str, "tcp://", 6) == 0) {
697  str += 6;
698  }
699 
700  if (sscanf(str, "%u.%u.%u.%u:%u%n", &a, &b, &c, &d, &port, &len) == 5) {
701  // Bind to a specific IPv4 address, e.g. 192.168.1.5:8080
702  sa->sin.sin_addr.s_addr = htonl((a << 24) | (b << 16) | (c << 8) | d);
703  sa->sin.sin_port = htons((uint16_t) port);
704 #ifdef NS_ENABLE_IPV6
705  } else if (sscanf(str, "[%99[^]]]:%u%n", buf, &port, &len) == 2 &&
706  inet_pton(AF_INET6, buf, &sa->sin6.sin6_addr)) {
707  // IPv6 address, e.g. [3ffe:2a00:100:7031::1]:8080
708  sa->sin6.sin6_family = AF_INET6;
709  sa->sin6.sin6_port = htons((uint16_t) port);
710 #endif
711  } else if (sscanf(str, "%199[^ :]:%u%n", host, &port, &len) == 2) {
712  sa->sin.sin_port = htons((uint16_t) port);
713  ns_resolve2(host, &sa->sin.sin_addr);
714  } else if (sscanf(str, "%u%n", &port, &len) == 1) {
715  // If only port is specified, bind to IPv4, INADDR_ANY
716  sa->sin.sin_port = htons((uint16_t) port);
717  }
718 
719  if (*use_ssl && (sscanf(str + len, ":%99[^:,]:%99[^:,]%n", cert, ca, &n) == 2 ||
720  sscanf(str + len, ":%99[^:,]%n", cert, &n) == 1)) {
721  len += n;
722  }
723 
724  return port < 0xffff && str[len] == '\0' ? len : 0;
725 }

931 static void ns_read_from_socket(struct ns_connection *conn) {
932  char buf[NS_READ_BUFFER_SIZE];
933  int n = 0;
934 
935  if (conn->flags & NSF_CONNECTING) {
936  int ok = 1, ret;
937  socklen_t len = sizeof(ok);
938 
939  ret = getsockopt(conn->sock, SOL_SOCKET, SO_ERROR, (char *) &ok, &len);
940  (void) ret;
941 #ifdef NS_ENABLE_SSL
942  if (ret == 0 && ok == 0 && conn->ssl != NULL) {
943  int res = SSL_connect(conn->ssl);
944  int ssl_err = ns_ssl_err(conn, res);
945  if (res == 1) {
946  conn->flags |= NSF_SSL_HANDSHAKE_DONE;
947  } else if (ssl_err == SSL_ERROR_WANT_READ ||
948  ssl_err == SSL_ERROR_WANT_WRITE) {
949  return; // Call us again
950  } else {
951  ok = 1;
952  }
953  conn->flags &= ~(NSF_WANT_READ | NSF_WANT_WRITE);
954  }
955 #endif
956  conn->flags &= ~NSF_CONNECTING;
957  DBG(("%p ok=%d", conn, ok));
958  if (ok != 0) {
959  conn->flags |= NSF_CLOSE_IMMEDIATELY;
960  }
961  ns_call(conn, NS_CONNECT, &ok);
962  return;
963  }
964 
965 #ifdef NS_ENABLE_SSL
966  if (conn->ssl != NULL) {
967  if (conn->flags & NSF_SSL_HANDSHAKE_DONE) {
968  // SSL library may have more bytes ready to read then we ask to read.
969  // Therefore, read in a loop until we read everything. Without the loop,
970  // we skip to the next select() cycle which can just timeout.
971  while ((n = SSL_read(conn->ssl, buf, sizeof(buf))) > 0) {
972  DBG(("%p %d <- %d bytes (SSL)", conn, conn->flags, n));
973  iobuf_append(&conn->recv_iobuf, buf, n);
974  ns_call(conn, NS_RECV, &n);
975  }
976  ns_ssl_err(conn, n);
977  } else {
978  int res = SSL_accept(conn->ssl);
979  int ssl_err = ns_ssl_err(conn, res);
980  if (res == 1) {
981  conn->flags |= NSF_SSL_HANDSHAKE_DONE;
982  conn->flags &= ~(NSF_WANT_READ | NSF_WANT_WRITE);
983  } else if (ssl_err == SSL_ERROR_WANT_READ ||
984  ssl_err == SSL_ERROR_WANT_WRITE) {
985  return; // Call us again
986  } else {
987  conn->flags |= NSF_CLOSE_IMMEDIATELY;
988  }
989  return;
990  }
991  } else
992 #endif
993  {
994  while ((n = (int) recv(conn->sock, buf, sizeof(buf), 0)) > 0) {
995  DBG(("%p %d <- %d bytes (PLAIN)", conn, conn->flags, n));
996  iobuf_append(&conn->recv_iobuf, buf, n);
997  ns_call(conn, NS_RECV, &n);
998  }
999  }
1000 
1001  if (ns_is_error(n)) {
1002  conn->flags |= NSF_CLOSE_IMMEDIATELY;
1003  }
1004 }

1076 time_t ns_mgr_poll(struct ns_mgr *mgr, int milli) {
1077  struct ns_connection *conn, *tmp_conn;
1078  struct timeval tv;
1079  fd_set read_set, write_set;
1080  sock_t max_fd = INVALID_SOCKET;
1081  time_t current_time = time(NULL);
1082 
1083  FD_ZERO(&read_set);
1084  FD_ZERO(&write_set);
1085  ns_add_to_set(mgr->ctl[1], &read_set, &max_fd);
1086 
1087  for (conn = mgr->active_connections; conn != NULL; conn = tmp_conn) {
1088  tmp_conn = conn->next;
1089  if (!(conn->flags & (NSF_LISTENING | NSF_CONNECTING))) {
1090  ns_call(conn, NS_POLL, &current_time);
1091  }
1092  if (conn->flags & NSF_CLOSE_IMMEDIATELY) {
1093  ns_close_conn(conn);
1094  } else {
1095  if (!(conn->flags & NSF_WANT_WRITE)) {
1096  //DBG(("%p read_set", conn));
1097  ns_add_to_set(conn->sock, &read_set, &max_fd);
1098  }
1099  if (((conn->flags & NSF_CONNECTING) && !(conn->flags & NSF_WANT_READ)) ||
1100  (conn->send_iobuf.len > 0 && !(conn->flags & NSF_CONNECTING) &&
1101  !(conn->flags & NSF_BUFFER_BUT_DONT_SEND))) {
1102  //DBG(("%p write_set", conn));
1103  ns_add_to_set(conn->sock, &write_set, &max_fd);
1104  }
1105  }
1106  }
1107 
1108  tv.tv_sec = milli / 1000;
1109  tv.tv_usec = (milli % 1000) * 1000;
1110 
1111  if (select((int) max_fd + 1, &read_set, &write_set, NULL, &tv) < 0) {
1112  return 0;
1113  } else {
1114  // select() might have been waiting for a long time, reset current_time
1115  // now to prevent last_io_time being set to the past.
1116  current_time = time(NULL);
1117 
1118  // Read wakeup messages
1119  if (mgr->ctl[1] != INVALID_SOCKET &&
1120  FD_ISSET(mgr->ctl[1], &read_set)) {
1121  struct ctl_msg ctl_msg;
1122  int len = (int) recv(mgr->ctl[1], (char *) &ctl_msg, sizeof(ctl_msg), 0);
1123  send(mgr->ctl[1], ctl_msg.message, 1, 0);
1124  if (len >= (int) sizeof(ctl_msg.callback) && ctl_msg.callback != NULL) {
1125  struct ns_connection *c;
1126  for (c = ns_next(mgr, NULL); c != NULL; c = ns_next(mgr, c)) {
1127  ctl_msg.callback(c, NS_POLL, ctl_msg.message);
1128  }
1129  }
1130  }
1131 
1132  for (conn = mgr->active_connections; conn != NULL; conn = tmp_conn) {
1133  tmp_conn = conn->next;
1134  if (FD_ISSET(conn->sock, &read_set)) {
1135  if (conn->flags & NSF_LISTENING) {
1136  if (conn->flags & NSF_UDP) {
1137  ns_handle_udp(conn);
1138  } else {
1139  // We're not looping here, and accepting just one connection at
1140  // a time. The reason is that eCos does not respect non-blocking
1141  // flag on a listening socket and hangs in a loop.
1142  accept_conn(conn);
1143  }
1144  } else {
1145  conn->last_io_time = current_time;
1146  ns_read_from_socket(conn);
1147  }
1148  }
1149 
1150  if (FD_ISSET(conn->sock, &write_set)) {
1151  if (conn->flags & NSF_CONNECTING) {
1152  ns_read_from_socket(conn);
1153  } else if (!(conn->flags & NSF_BUFFER_BUT_DONT_SEND)) {
1154  conn->last_io_time = current_time;
1155  ns_write_to_socket(conn);
1156  }
1157  }
1158  }
1159  }
1160 
1161  for (conn = mgr->active_connections; conn != NULL; conn = tmp_conn) {
1162  tmp_conn = conn->next;

1703 // A helper function for traversing a comma separated list of values.
1704 // It returns a list pointer shifted to the next value, or NULL if the end
1705 // of the list found.
1706 // Value is stored in val vector. If value has form "x=y", then eq_val
1707 // vector is initialized to point to the "y" part, and val vector length
1708 // is adjusted to point only to "x".
1709 static const char *next_option(const char *list, struct vec *val,
1710  struct vec *eq_val) {
1711  if (list == NULL || *list == '\0') {
1712  // End of the list
1713  list = NULL;
1714  } else {
1715  val->ptr = list;
1716  if ((list = strchr(val->ptr, ',')) != NULL) {
1717  // Comma found. Store length and shift the list ptr
1718  val->len = list - val->ptr;
1719  list++;
1720  } else {
1721  // This value is the last one
1722  list = val->ptr + strlen(val->ptr);
1723  val->len = list - val->ptr;
1724  }

1726  if (eq_val != NULL) {
